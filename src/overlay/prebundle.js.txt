function click(selector) {
    let found = document.querySelector(selector);
    found?.click();
};
function delay(duration) {
    return new Promise((r) => setTimeout(r, duration));
};
function hide(selector) {
    let found = document.querySelector(selector);
    if (found) {
        found.style.display = "none";
    }
};
